function fizzBuzzTree(tree) {
  let _walk = (node) => {
    if(node.left){
      _walk(node.left);
    }

      _walk(node.right);
    }

    if(node.value % 3 === 0){

  };

}
 
function findMax(tree){
  let max = 0;
  const _walk = (node) => {

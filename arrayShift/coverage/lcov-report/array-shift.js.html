"use strict"
 
const ar = [1,2,3,4,5];
const object = { "bob": "is name", "age": "is old"};
const string = "string";
 
const insertShiftArray = (array,value) => {
    if(Array.isArray(array) === false){
        return "not an array";
    }
 
    let newArr = [];
 
    for(let i = 0; i <= array.length-1;i++){
        newArr[i] = array[i];
    }
 
    let center = Math.ceil(newArr.length/2 - 1);
    for(let i = newArr.length-1; i >= center; i--){
        newArr[i+1] = newArr[i];
    }
    newArr[center] = value;
    return newArr;
}
 
module.exports = insertShiftArray;
 
console.log(insertShiftArray(ar,2));
console.log(insertShiftArray(string,2));
console.log(insertShiftArray(ar,object));

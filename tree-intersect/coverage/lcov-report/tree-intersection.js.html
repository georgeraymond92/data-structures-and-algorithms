function intersection(treeA,treeB) {
  let seenValues = {};
  let duplicates = {};
 
  const _go = (node,cb) => {
    let current = node;
    if(current.left){
      _go(current.left,cb)
    }
    if(current.right){
      _go(current.right, cb);
    }
    cb(seenValues,current.value, duplicates);
  }
